\section{Exponenciální funkce}
\begin{definition}
    Nechť $a\in \mathbb R^+ \smallsetminus \left \{ 1 \right \}. $ Pak funkci $f\colon y=a^x$
    nazýváme \textbf{exponenciální funkcí} o základu $a$. Funkci $f\colon y=e^x$ pak
    nazveme \textbf{přirozenou exponenciální funkcí}.
\end{definition}

\begin{veta}
    Vlastnosti exponenciálních funkcí $f\colon y= a^x, a \in \mathbb N$:
    \begin{enumerate}[$i.$]
        \item $\mathcal D(f)= \mathbb R$, $\mathcal H(f)= \mathbb R^+$;
       	\item graf každé exponenciální funkce prochází bodem $[0,1]$;
        \item $f$ není ani sudá, ani lichá;
        \item $f$ je zdola omezená;
        \item pro $a >1$ je $f$ rostoucí, pro $a<1$ je klesající;
        \item $f$ nemá extrémy a není periodická.
    \end{enumerate}
\end{veta}

\begin{definition}
    (Ne)rovnice s neznámou v exponentu se nazývá \textbf{exponenciální (ne)rovnice}.
\end{definition}

\begin{veta}
    Pro všechna $a \in \mathbb R^+ \smallsetminus \left \{ 1 \right \},$ a $x_1, x_2
    \in \mathbb R$ platí $a^{x_1}=a^{x_2}$ právě tehdy, když $x_1=x_2.$
\end{veta}

\begin{proof}
    Plyne z toho, že exponenciální funkce je prostá.
\end{proof}

\begin{priklad}
Určete, pro která $b\in \mathbb R\smallsetminus\left \{ 1 \right \} $ je $f\colon y=\left ( \frac{b}{b-1} \right )^x $
rostoucí a pro která klesající.
\end{priklad}

\begin{reseni}
Je rostoucí, pokud základ je větší než 1 a klesající, pokud základ je menší než 1.
Řešíme tedy
\begin{align*}
\frac{b}{b-1}>1, & & \textrm{resp.} & & \frac{b}{b-1}<1.
\end{align*}
\end{reseni}
