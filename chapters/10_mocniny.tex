\section{Mocniny, mocninná funkce}

\begin{definition}
    Nechť $n\in \mathbb N.$ Pak funkci $f\colon y=x^n$ nazýváme \textbf{mocninnou funkcí}
    s~přirozeným exponentem.
\end{definition}

\begin{definition}
    Nechť $n\in \mathbb N.$ Pak funkci $f\colon y=x^{-n}=\frac{1}{x^n}$ nazýváme
    \textbf{mocninnou funkcí} s~celým záporným exponentem.
\end{definition}

\begin{definition}
    Nechť $\frac{p}{q}\in \mathbb Q, x \in \mathbb R^+.$ Pak funkci $f\colon y=
    x^{\frac{p}{q}}=\sqrt[q]{x^p}$ nazýváme \textbf{mocninnou funkcí}
    s racionálním exponentem.
\end{definition}

\begin{veta}[Pravidla pro počítání s mocninami]
    Pro všechna $a \in \mathbb R, m,n \in \mathbb N$ platí
    \begin{enumerate}[$i.$]
        \item $a^m\cdot a^n=a^{m+n},$
       	\item $(ab)^n = a^n\cdot b^n,$
       	\item $a^m : a^n = a^{m-n}, a \ne 0,$
       	\item $(a : b)^n = a^n : b^n, b \ne 0,$
       	\item $\left ( a^m \right )^n = a^{mn}.$
    \end{enumerate}
\end{veta}

\begin{proof}
    Zřejmé.
\end{proof}

\begin{veta}
  Vlastnosti mocninných funkcí $f\colon y= x^n, n \in \mathbb N$:
  \begin{enumerate}[1.]
    \item Pokud $n$ je liché
    \begin{enumerate}[$i.$]
      \item $\mathcal D(f)= \mathbb R$, $\mathcal H(f)= \mathbb R$;
      \item $f$ je lichá;
      \item $f$ není omezená shora ani zdola;
      \item $f$ je rostoucí v celém svém definičním oboru;
      \item $f$ nemá ani maximum, ani minimum.
    \end{enumerate}
    \item Pokud $n$ je sudé
    \begin{enumerate}[$i.$]
      \item $\mathcal D(f)= \mathbb R$, $\mathcal H(f)= \left<  0,+\infty \right)$;
      \item $f$ je sudá;
      \item $f$ není omezená shora, je omezená zdola;
      \item $f$ je rostoucí v intervalu $\left<  0,+\infty \right)$, je klesající v intervalu $\left( -\infty,0 \right> $;
      \item $f$ nemá maximum, má minimum v bodě $0$.
    \end{enumerate}
  \end{enumerate}
\end{veta}

\begin{veta}
  Vlastnosti mocninných funkcí $f\colon y= x^{-n}, n \in \mathbb N$:
  \begin{enumerate}[1.]
    \item Pokud $n$ je liché
    \begin{enumerate}[$i.$]
      \item $\mathcal D(f)= \mathbb R \smallsetminus \{ 0\}$, $\mathcal H(f)= \mathbb R \smallsetminus \{ 0 \}$;
      \item $f$ je lichá;
      \item $f$ není omezená shora ani zdola;
      \item $f$ je klesající v $( -\infty,0)$ a v $( 0,+\infty )$;
      \item $f$ nemá maximum, ani minimum.
    \end{enumerate}
    \item Pokud $n$ je sudé
    \begin{enumerate}[$i.$]
      \item $\mathcal D(f)= \mathbb R \smallsetminus \{ 0\}$, $\mathcal H(f)= ( 0,+\infty )$;
      \item $f$ je sudá;
      \item $f$ není omezená shora, je zdola omezená;
      \item $f$ je rostoucí v intervalu $( -\infty,0 ) $, je klesající v intervalu $( 0,+\infty )$;
      \item $f$ nemá maximum ani minimum.
    \end{enumerate}
  \end{enumerate}
\end{veta}

\begin{definition}
    Nechť $n\in \mathbb N, n\geq 2, x \geq 0.$ Pak inverzní funkci k funkci
    $f\colon y=x^n$ nazýváme \textbf{$n$-tou odmocninou} a značíme $f^{-1}\colon y=\sqrt[n]{x}.$
\end{definition}

\begin{veta}[Pravidla pro počítání s odmocninami]
    Pro všechna $a,b \in \mathbb R_0^+ m,n \in \mathbb N$ platí
    \begin{enumerate}[$i.$]
        \item $\sqrt[m]{a}\cdot \sqrt[m]{b}=\sqrt[m]{ab},$
       	\item $\sqrt[m]{\sqrt[n]{a} } = \sqrt[mn]{a},$
       	\item $\sqrt[m]{a} :\sqrt[m]{b} = \sqrt[m]{a:b},$
       	\item $\left ( \sqrt[m]{a}  \right )^n = \sqrt[m]{a^n},$
       	\item $\forall k \in \mathbb N\colon \sqrt[m]{a}=\sqrt[km]{a^k}.$
    \end{enumerate}
\end{veta}

\begin{proof}
    Zřejmé.
\end{proof}

\begin{veta}
    Vlastnosti odmocninných funkcí $f\colon y= \sqrt[n]{x}, n \in \mathbb N, n > 1$:
    \begin{enumerate}[$i.$]
        \item $f$ není ani lichá, ani sudá;
       	\item $f$ je rostoucí;
        \item $f$ je zdola omezená;
        \item $f$ má ostré minimum v bodě $0$;
        \item $f$ není periodická.
    \end{enumerate}
\end{veta}
